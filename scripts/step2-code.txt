        (async () => {
          try {
            // パスワード入力欄を取得
            const passwordInput = document.querySelector('input[name="current_password"][type="password"]');
            
            if (!passwordInput) {
              return { didAction: false, reason: 'パスワード入力フィールドが見つかりませんでした' };
            }
            
            const password = '{{account_password}}';
            
            // フォーカス & クリック（視覚的な意味合いは薄いが一応）
            passwordInput.focus();
            passwordInput.click();
            await new Promise(r => setTimeout(r, 200));
            
            // 値をクリアしてから入力
            passwordInput.value = '';
            
            // React対応: HTMLInputElementのネイティブsetterを使用
            const setter = Object.getOwnPropertyDescriptor(HTMLInputElement.prototype, 'value').set;
            setter.call(passwordInput, password);
            passwordInput.dispatchEvent(new Event('input', { bubbles: true }));
            passwordInput.dispatchEvent(new Event('change', { bubbles: true }));
            await new Promise(r => setTimeout(r, 200));
            
            // --- Enterイベントは使わず、Confirmボタンをクリックする ---
            
            // 画面上のボタンを全部取得して「Confirm」と書かれたものを探す
            const buttons = Array.from(document.querySelectorAll('button, [role="button"]'));
            const confirmButton = buttons.find(btn => {
              const text = (btn.textContent || '').trim();
              return /confirm/i.test(text);
            });
            
            if (!confirmButton) {
              return { didAction: false, reason: 'Confirmボタンが見つかりませんでした' };
            }
            
            // Confirmボタンクリック
            confirmButton.click();
            await new Promise(r => setTimeout(r, 500));
            
            return { didAction: true, reason: 'パスワードを入力してConfirmボタンをクリックしました' };
          } catch (e) {
            return { didAction: false, reason: 'エラー: ' + String(e) };
          }
        })()
